cmake note

本文是cmake实践的笔记

基本构建

PROJECT (HELLO)
SET(SRC_LIST main.c)
MESSAGE(STATUS "This is BINARY dir " ${HELLO_BINARY_DIR}) 
MESSAGE(STATUS "This is SOURCE dir "${HELLO_SOURCE_DIR}) 
ADD_EXECUTABLE(hello ${SRC_LIST})
  1. PROJECT(projectname [CXX] [C] [Java]):工程名
  2. SET (VAR [VALUE] [CACHE TYPE DOCSTRING [FORCE]])：设定变量

  3. MESSAGE([SEND_ERROR | STATUS | FATAL_ERROR] "message to display" ...)

    SEND_ERROR:产生错误，生成过程被跳过
    STATUS: 输出前缀为--的信息
    FATAL_ERROR:立即终止所有cmake过程

  4. ADD_EXECUTABLE(hello ${SRC_LIST}):定义了这个工程会生成一个文件名为 hello的可执行文件,相关的源文件是 SRC_LIST 中定义的源文件列表
  5. cmake 中除了在if中之外，引用变量需要用${VAR}括起来

CMakeLists.txt

工程CMakeLists.txt

PROJECT (HELLO CXX)
MESSAGE(STATUS "This is Binary dir" ${HELLO_BINARY_DIR})
MESSAGE(STATUS "This is SOURCE dir" ${HELLO_SOURCE_DIR})
install(FILES COPYRIGHT README DESTINATION share/doc/cmake/t2)
INSTALL(PROGRAMS runhello.sh DESTINATION bin)
install(DIRECTORY doc/ DESTINATION share/doc/cmake/t2)
ADD_SUBDIRECTORY(src bin)

src下的CMakeLists.txt

set(EXECUTABLE_OUTPUT_PATH ${PROJECT_BINARY_DIR}/bin)
ADD_EXECUTABLE(hello main.cpp)
INSTALL(TARGETS hello
    RUNTIME DESTINATION bin
    )

说明

  1. ADD_SUBDIRECTORY(source_dir [binary_dir] [EXCLUDE_FROM_ALL]):这个指令用于向当前工程添加存放源文件的子目录,并可以指定中间二进制和目标二进制存放的位置。EXCLUDE_FROM_ALL 参数的含义是将这个目录从编译过程中排除.指定 bin 目录后,相当于在编译时 将 src 重命名为 bin,所有的中间结果和目标二进制都将存放在 bin 目录。
  2. SET(EXECUTABLE_OUTPUT_PATH ${PROJECT_BINARY_DIR}/bin):换个位置保存目标二进制

  3. SET(LIBRARY_OUTPUT_PATH ${PROJECT_BINARY_DIR}/lib)

    问题是,我应该把这两条指令写在工程的 CMakeLists.txt 还是 src 目录下的 CMakeLists.txt,把握一个简单的原则,在哪里 ADD_EXECUTABLE 或 ADD_LIBRARY, 如果需要改变目标存放路径,就在哪里加入上述的定义。

安装

安装前缀在通过-DCMAKE_INSTALL_PREFIX指定，如:cmake -DCMAKE_INSTALL_PREFIX=/share/usr ..如果没有指定前缀，默认安装到/usr/local,
在CMakeLists.txt中使用DESTINATION指代它

  1. 目标可执行文件的安装
  INSTALL(TARGETS targets...
            [[ARCHIVE|LIBRARY|RUNTIME]
                        [DESTINATION ]
                        [PERMISSIONS permissions...]
                        [CONFIGURATIONS
          [Debug|Release|...]]
                        [COMPONENT ]
                        [OPTIONAL]
                       ] [...])

例子

  INSTALL(TARGETS myrun mylib mystaticlib
     RUNTIME DESTINATION bin
     LIBRARY DESTINATION lib
     ARCHIVE DESTINATION libstatic
  )

其中myrun为add_executable中添加的可执行文件变量

  1. 普通文件的安装
  INSTALL(FILES files... DESTINATION 
            [PERMISSIONS permissions...]
            [CONFIGURATIONS [Debug|Release|...]]
            [COMPONENT ]
            [RENAME ] [OPTIONAL])

默认权限为644

  1. 非目标文件的可执行文件安装(比如shell脚本之类)
  INSTALL(PROGRAMS files... DESTINATION 
            [PERMISSIONS permissions...]
            [CONFIGURATIONS [Debug|Release|...]]
            [COMPONENT ]
            [RENAME ] [OPTIONAL])

默认755权限

  1. 目录的安装

    INSTALL(DIRECTORY dirs... DESTINATION <dir>
         [FILE_PERMISSIONS permissions...]
         [DIRECTORY_PERMISSIONS permissions...]
         [USE_SOURCE_PERMISSIONS]
         [CONFIGURATIONS [Debug|Release|...]]
         [COMPONENT <component>]
         [[PATTERN <pattern> | REGEX <regex>]
          [EXCLUDE] [PERMISSIONS permissions...]] [...])

    如果dirs以/结尾，则安装dirs下的所有文件，否则连同父目录一同安装
    使用PATTERN可以使用正则表达式过滤

    
INSTALL(DIRECTORY icons scripts/ DESTINATION share/myproj
PATTERN "CVS" EXCLUDE
         PATTERN "scripts/*"
         PERMISSIONS OWNER_EXECUTE OWNER_WRITE OWNER_READ
                 GROUP_EXECUTE GROUP_READ)

  2. 安装时CMAKE脚本执行

    安装时 CMAKE 脚本的执行:
INSTALL([[SCRIPT <file>] [CODE <code>]] [...])
SCRIPT 参数用于在安装时调用 cmake 脚本文件(也就是<abc>.cmake 文件) CODE 参数用于执行 CMAKE 指令,必须以双引号括起来。比如: INSTALL(CODE "MESSAGE(\"Sample install message.\")")

静态库和动态库的构建

目录结构

.
├── CMakeLists.txt
├── build
└── lib
    ├── CMakeLists.txt
    ├── hello.cpp
    └── hello.h

CMakeLists.txt

工程CMakeLists.txt

PROJECT(HELLOLIB)
ADD_SUBDIRECTORY(lib)

lib/CMakeLists.txt

SET(LIB_SRC hello.cpp)
ADD_LIBRARY(hello SHARED ${LIB_SRC})
ADD_LIBRARY(hello_static STATIC ${LIB_SRC})
SET_TARGET_PROPERTIES(hello_static PROPERTIES OUTPUT_NAME hello)
GET_TARGET_PROPERTY(OUTPUT_VALUE hello_static OUTPUT_NAME)
MESSAGE(STATUS "The output name of hello_static is:" ${OUTPUT_VALUE})
SET_TARGET_PROPERTIES(hello PROPERTIES CLEAN_DIRECT 1)
SET_TARGET_PROPERTIES(hello_static PROPERTIES CLEAN_DIRECT 1)
SET_TARGET_PROPERTIES(hello PROPERTIES VERSION 1.1 SOVERSION 1)

INSTALL(TARGETS hello hello_static
    LIBRARY DESTINATION lib
    ARCHIVE DESTINATION lib
    )

INSTALL(FILES hello.h DESTINATION include)

解释

  1. ADD_LIBRARY

    ADD_LIBRARY(libname    [SHARED|STATIC|MODULE]
       [EXCLUDE_FROM_ALL]
             source1 source2 ... sourceN)

    SHARED: 动态链接库
    STATIC: 静态链接库
    MODULE: 在使用 dyld 的系统有效,如果不支持 dyld,则被当作 SHARED 对待
    EXCLUDE_FROM_ALL 参数的意思是这个库不会被默认构建,除非有其他的组件依赖或者手 工构建。

  2. SET_TARGET_PROPERTIES

    SET_TARGET_PROPERTIES,其基本语法是: SET_TARGET_PROPERTIES(target1 target2 ...
                   PROPERTIES prop1 value1
                   prop2 value2 ...)

  3. 版本号

    SET_TARGET_PROPERTIES(hello PROPERTIES VERSION 1.2 SOVERSION 1)

  • VERSION: 动态库版本
  • SOVERSION: API版本

头文件添加

如果共享库头文件不在标准搜索路径/usr/include /usr/local/include则需要手动添加

  1. INCLUDE_DIRECTORIES([AFTER|BEFORE] [SYSTEM] dir1 dir2 ...)
    SET(CMAKE_INCLUDE_DIRECTORIES_BEFORE on):将添加的头文件路径放在已有路径前面
    INCLUDE_DIRECTORIES(/usr/include/hello)

  2. 通过设置环境变量

    export CMAKE_INCLUDE_PATH=/usr/include/hello
    然后将INCLUDE_DIRECTORIES(/usr/include/hello)替换为

    FIND_PATH(myHeader hello.h)
IF(myHeader)
INCLUDE_DIRECTORIES(${myHeader})
ENDIF(myHeader)

    FIND_PATH格式:FIND_PATH(myHeader NAMES hello.h PATHS /usr/include /usr/include/hello)
